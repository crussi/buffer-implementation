static ActionStack *stack;

TEST_GROUP(ACTION_STACK);

TEST_SETUP(ACTION_STACK)
{
    stack = new_action_stack(0, 0);
}

TEST_TEAR_DOWN(ACTION_STACK)
{
    free_action_stack(stack);
    stack = NULL;
}

// ---------------------------------------------------------------------------
// Helpers
// ---------------------------------------------------------------------------

static Action make_char_action(ActionType type, int row, int col, char c) {
    Action a;
    a.type         = type;
    a.position.row = row;
    a.position.col = col;
    a.character    = c;
    a.text         = NULL;
    return a;
}

static Action make_text_action(ActionType type, int row, int col,
                               const char *text) {
    Action a;
    a.type         = type;
    a.position.row = row;
    a.position.col = col;
    a.character    = '\0';
    a.text         = text ? strdup(text) : NULL;
    return a;
}

TEST(ACTION_STACK, new_action_stack_returns_non_null) {
    TEST_ASSERT_NOT_NULL(stack);
}

TEST(ACTION_STACK, new_action_stack_is_empty) {
    TEST_ASSERT_TRUE(action_stack_is_empty(stack));
}

TEST(ACTION_STACK, new_action_stack_has_zero_size) {
    TEST_ASSERT_EQUAL_size_t(0, stack->size);
}

TEST(ACTION_STACK, new_action_stack_custom_initial_capacity) {
    ActionStack *s = new_action_stack(64, 0);
    TEST_ASSERT_NOT_NULL(s);
    TEST_ASSERT_EQUAL_size_t(64, s->capacity);
    free_action_stack(s);
}

TEST(ACTION_STACK, new_action_stack_max_capacity_clamps_initial) {
    ActionStack *s = new_action_stack(128, 8);
    TEST_ASSERT_NOT_NULL(s);
    TEST_ASSERT_EQUAL_size_t(8, s->capacity);
    free_action_stack(s);
}

// ---------------------------------------------------------------------------
// push
// ---------------------------------------------------------------------------

TEST(ACTION_STACK, push_returns_true_on_success) {
    Action a = make_char_action(INSERT_CHAR, 0, 0, 'x');
    TEST_ASSERT_TRUE(push_action(stack, a));
}

TEST(ACTION_STACK, push_increases_size) {
    Action a = make_char_action(INSERT_CHAR, 0, 0, 'x');
    push_action(stack, a);
    TEST_ASSERT_EQUAL_size_t(1, stack->size);
}

TEST(ACTION_STACK, push_multiple_increases_size) {
    for (int i = 0; i < 10; i++) {
        Action a = make_char_action(INSERT_CHAR, 0, i, (char)('a' + i));
        push_action(stack, a);
    }
    TEST_ASSERT_EQUAL_size_t(10, stack->size);
}

TEST(ACTION_STACK, push_null_stack_returns_false) {
    Action a = make_char_action(INSERT_CHAR, 0, 0, 'x');
    TEST_ASSERT_FALSE(push_action(NULL, a));
}

TEST(ACTION_STACK, push_triggers_growth_beyond_default_capacity) {
    for (int i = 0; i < 32; i++) {
        Action a = make_char_action(INSERT_CHAR, 0, i, 'a');
        TEST_ASSERT_TRUE(push_action(stack, a));
    }
    TEST_ASSERT_EQUAL_size_t(32, stack->size);
    TEST_ASSERT_TRUE(stack->capacity >= 32);
}

// ---------------------------------------------------------------------------
// pop
// ---------------------------------------------------------------------------

TEST(ACTION_STACK, pop_returns_false_on_empty_stack) {
    Action out;
    TEST_ASSERT_FALSE(pop_action(stack, &out));
}

TEST(ACTION_STACK, pop_returns_true_when_item_present) {
    Action a = make_char_action(INSERT_CHAR, 1, 2, 'z');
    push_action(stack, a);
    Action out;
    TEST_ASSERT_TRUE(pop_action(stack, &out));
}

TEST(ACTION_STACK, pop_decreases_size) {
    Action a = make_char_action(INSERT_CHAR, 0, 0, 'a');
    push_action(stack, a);
    Action out;
    pop_action(stack, &out);
    TEST_ASSERT_EQUAL_size_t(0, stack->size);
}

TEST(ACTION_STACK, pop_returns_correct_action) {
    Action a = make_char_action(DELETE_CHAR, 3, 7, 'q');
    push_action(stack, a);
    Action out;
    pop_action(stack, &out);
    TEST_ASSERT_EQUAL_INT(DELETE_CHAR,  out.type);
    TEST_ASSERT_EQUAL_INT(3,            out.position.row);
    TEST_ASSERT_EQUAL_INT(7,            out.position.col);
    TEST_ASSERT_EQUAL_CHAR('q',         out.character);
}

TEST(ACTION_STACK, pop_is_lifo_order) {
    for (int i = 0; i < 5; i++) {
        Action a = make_char_action(INSERT_CHAR, 0, i, (char)('a' + i));
        push_action(stack, a);
    }
    for (int i = 4; i >= 0; i--) {
        Action out;
        pop_action(stack, &out);
        TEST_ASSERT_EQUAL_CHAR((char)('a' + i), out.character);
    }
}

TEST(ACTION_STACK, pop_null_stack_returns_false) {
    Action out;
    TEST_ASSERT_FALSE(pop_action(NULL, &out));
}

TEST(ACTION_STACK, peek_returns_top_item) {
    Action a = make_char_action(INSERT_CR,  2, 0, '\0');
    Action b = make_char_action(DELETE_CR,  5, 0, '\0');
    push_action(stack, a);
    push_action(stack, b);
    Action out;
    peek_action(stack, &out);
    TEST_ASSERT_EQUAL_INT(DELETE_CR, out.type);
    TEST_ASSERT_EQUAL_INT(5,         out.position.row);

TEST(ACTION_STACK, reset_clears_size) {
    for (int i = 0; i < 5; i++) {
        Action a = make_char_action(INSERT_CHAR, 0, i, 'a');
        push_action(stack, a);
    }
    reset_action_stack(stack);
    TEST_ASSERT_EQUAL_size_t(0, stack->size);
}

TEST(ACTION_STACK, reset_preserves_capacity) {
    for (int i = 0; i < 5; i++) {
        Action a = make_char_action(INSERT_CHAR, 0, i, 'a');
        push_action(stack, a);
    }
    size_t cap = stack->capacity;
    reset_action_stack(stack);
    TEST_ASSERT_EQUAL_size_t(cap, stack->capacity);
}

TEST(ACTION_STACK, reset_then_push_works) {
    Action a = make_char_action(INSERT_CHAR, 0, 0, 'r');
    push_action(stack, a);
    reset_action_stack(stack);
    Action b = make_char_action(INSERT_CHAR, 0, 0, 's');
    TEST_ASSERT_TRUE(push_action(stack, b));
    TEST_ASSERT_EQUAL_size_t(1, stack->size);
}

TEST(ACTION_STACK, reset_null_does_not_crash) {
    reset_action_stack(NULL);
}

TEST(BUFFER, newBuf_returns_non_null)
{
    buffer *buf = newBuf();
    TEST_ASSERT_NOT_NULL(buf);
    freeBuf(buf);
}

TEST(BUFFER, newBuf_has_one_empty_row)
{
    buffer *buf = newBuf();
    TEST_ASSERT_EQUAL_INT(1, buf->numrows);
    TEST_ASSERT_EQUAL_INT(0, buf->rows[0].length);
    TEST_ASSERT_EQUAL_STRING("", buf->rows[0].line);
    freeBuf(buf);
}

// ---------------------------------------------------------------------------
// insertChar
// ---------------------------------------------------------------------------

TEST(BUFFER, insertChar_appends_to_empty_row)
{
    buffer *buf = newBuf();
    insertChar(&buf->rows[0], 0, 'H');
    TEST_ASSERT_EQUAL_INT(1, buf->rows[0].length);
    TEST_ASSERT_EQUAL_CHAR('H', buf->rows[0].line[0]);
    freeBuf(buf);
}

TEST(BUFFER, insertChar_builds_string)
{
    buffer *buf = newBuf();
    const char *word = "Hello";
    for (int i = 0; i < 5; i++)
        insertChar(&buf->rows[0], i, word[i]);
    TEST_ASSERT_EQUAL_STRING("Hello", buf->rows[0].line);
    TEST_ASSERT_EQUAL_INT(5, buf->rows[0].length);
    freeBuf(buf);
}

TEST(BUFFER, insertChar_inserts_in_middle)
{
    buffer *buf = newBuf();
    insertChar(&buf->rows[0], 0, 'A');
    insertChar(&buf->rows[0], 1, 'C');
    insertChar(&buf->rows[0], 1, 'B');
    TEST_ASSERT_EQUAL_STRING("ABC", buf->rows[0].line);
    freeBuf(buf);
}

TEST(BUFFER, insertChar_clamps_negative_index)
{
    buffer *buf = newBuf();
    insertChar(&buf->rows[0], 0, 'X');
    insertChar(&buf->rows[0], -5, 'Y');
    TEST_ASSERT_EQUAL_CHAR('Y', buf->rows[0].line[0]);
    TEST_ASSERT_EQUAL_CHAR('X', buf->rows[0].line[1]);
    freeBuf(buf);
}

TEST(BUFFER, insertChar_clamps_past_end)
{
    buffer *buf = newBuf();
    insertChar(&buf->rows[0], 0, 'A');
    insertChar(&buf->rows[0], 100, 'B');
    TEST_ASSERT_EQUAL_STRING("AB", buf->rows[0].line);
    freeBuf(buf);
}

TEST(BUFFER, bufToFile_roundtrip)
{
    buffer *buf = newBuf();
    const char *words[] = { "alpha", "beta", "gamma" };

    for (int i = 0; i < (int)strlen(words[0]); i++)
        insertChar(&buf->rows[0], i, words[0][i]);

    insertCR(buf, 0, buf->rows[0].length);
    for (int i = 0; i < (int)strlen(words[1]); i++)
        insertChar(&buf->rows[1], i, words[1][i]);

    insertCR(buf, 1, buf->rows[1].length);
    for (int i = 0; i < (int)strlen(words[2]); i++)
        insertChar(&buf->rows[2], i, words[2][i]);

    FILE *f = bufToFile(buf);
    TEST_ASSERT_NOT_NULL(f);

    buffer *buf2 = fileToBuf(f);
    fclose(f);

    TEST_ASSERT_EQUAL_INT(3, buf2->numrows);
    TEST_ASSERT_EQUAL_STRING("alpha", buf2->rows[0].line);
    TEST_ASSERT_EQUAL_STRING("beta",  buf2->rows[1].line);
    TEST_ASSERT_EQUAL_STRING("gamma", buf2->rows[2].line);

    freeBuf(buf);
    freeBuf(buf2);
}

TEST(BUFFER, bufToFile_null_buf_returns_null)
{
    FILE *f = bufToFile(NULL);

// Mixed sequences
TEST(EDITOR_HISTORY, undo_sequence_restores_in_reverse_order) {
    insertChar(&buf->rows[0], 0, 'h');
    history_record(h, make_insert_char(0, 0, 'h'));

TEST(EDITOR_HISTORY, new_edit_after_undo_clears_redo) {

    TEST_ASSERT_EQUAL_size_t(1, h->redo_stack->size);

    TEST_ASSERT_EQUAL_size_t(0, h->redo_stack->size);

TEST(EDITOR_HISTORY, redo_is_unavailable_after_new_edit) {
    insertChar(&buf->rows[0], 0, 'a');

    history_undo(h, buf);

    insertChar(&buf->rows[0], 0, 'b');
    history_record(h, make_insert_char(0, 0, 'b'));

    TEST_ASSERT_FALSE(history_redo(h, buf));
}

TEST(EDITOR_HISTORY, full_undo_redo_cycle_mixed_actions) {
    insertChar(&buf->rows[0], 0, 'a');
    history_record(h, make_insert_char(0, 0, 'a'));
    insertChar(&buf->rows[0], 1, 'b');
    history_record(h, make_insert_char(0, 1, 'b'));
    insertCR(buf, 0, 2);
    history_record(h, make_insert_cr(0, 2));
    insertChar(&buf->rows[1], 0, 'c');
    history_record(h, make_insert_char(1, 0, 'c'));

    TEST_ASSERT_EQUAL_INT(2, buf->numrows);
    TEST_ASSERT_EQUAL_STRING("ab", buf->rows[0].line);
    TEST_ASSERT_EQUAL_STRING("c",  buf->rows[1].line);

    history_undo(h, buf);
    TEST_ASSERT_EQUAL_STRING("", buf->rows[1].line);

    history_undo(h, buf);
    TEST_ASSERT_EQUAL_INT(1, buf->numrows);
    TEST_ASSERT_EQUAL_STRING("ab", buf->rows[0].line);

    history_undo(h, buf);
    TEST_ASSERT_EQUAL_STRING("a", buf->rows[0].line);

    history_undo(h, buf);
    TEST_ASSERT_EQUAL_INT(0, buf->rows[0].length);

    history_redo(h, buf);
    history_redo(h, buf);
    history_redo(h, buf);
    history_redo(h, buf);

    TEST_ASSERT_EQUAL_INT(2, buf->numrows);
    TEST_ASSERT_EQUAL_STRING("ab", buf->rows[0].line);
    TEST_ASSERT_EQUAL_STRING("c",  buf->rows[1].line);
}

    }
